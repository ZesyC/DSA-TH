    int n = 10;        // số phần tử muốn tạo
    int L = -10, R = 10; // khoảng ngẫu nhiên [L, R]

    srand((unsigned)time(NULL)); // seed để mỗi lần chạy ra dãy khác

    Node* head = NULL;
    for (int i = 0; i < n; ++i){
        int x = L + rand() % (R - L + 1); // số ngẫu nhiên trong [L, R]
        push_back(&head, x);
    }
